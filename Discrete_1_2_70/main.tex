%--- Problem Statement ---
\begin{frame}{Problem}
\textbf{Solve the differential equation:}
\[
\frac{dy}{dx} = x^2 + 2y
\]
with initial condition $y = 1$ at $x = 0$.

\vspace{1em}
\textbf{Options:}
\begin{enumerate}[(a)]
    \item $y = \dfrac{2}{3}x^2 + 4x$
    \item $y = \sqrt{\dfrac{2}{3}x^3 + 4x + k}$
    \item $y = \dfrac{2}{3}x^3 - 4x + k$
    \item $y = \dfrac{2}{3}x^3 + 4x + k$
\end{enumerate}
\end{frame}

%--- Step 1 ---
\begin{frame}{Step 1: Rewrite in Standard Form}
Rearrange:
\[
\frac{dy}{dx} - 2y = x^2
\]

This is a \textbf{first-order linear ODE} of the form:
\[
\frac{dy}{dx} + P(x)\,y = Q(x)
\]
where
\[
P(x) = -2, \qquad Q(x) = x^2
\]
\end{frame}

%--- Step 2 ---
\begin{frame}{Step 2: Integrating Factor}
The integrating factor is:
\[
\mu(x) = e^{\int P(x)\,dx} = e^{\int -2\,dx} = e^{-2x}
\]

Multiply both sides by $\mu(x) = e^{-2x}$:
\[
e^{-2x}\frac{dy}{dx} - 2e^{-2x}y = x^2 e^{-2x}
\]

The left-hand side becomes a perfect derivative:
\[
\frac{d}{dx}\!\left(y\,e^{-2x}\right) = x^2 e^{-2x}
\]
\end{frame}

%--- Step 3 ---
\begin{frame}{Step 3: Integrate Both Sides}
\[
\int d\!\left(y\,e^{-2x}\right) = \int x^2\,e^{-2x}\,dx
\]

Using integration by parts twice on the right:
\[
\int x^2 e^{-2x}\,dx = -\frac{x^2}{2}e^{-2x} - \frac{x}{2}e^{-2x} - \frac{1}{4}e^{-2x} + C
\]

Therefore:
\[
y\,e^{-2x} = -\frac{x^2}{2}e^{-2x} - \frac{x}{2}e^{-2x} - \frac{1}{4}e^{-2x} + C
\]

Dividing by $e^{-2x}$:
\[
y = Ce^{2x} - \frac{x^2}{2} - \frac{x}{2} - \frac{1}{4}
\]
\end{frame}

%--- Step 4 ---
\begin{frame}{Step 4: Apply Initial Condition}
Apply $y = 1$ at $x = 0$:
\[
1 = C\,e^{0} - 0 - 0 - \frac{1}{4}
\]
\[
1 = C - \frac{1}{4} \implies C = \frac{5}{4}
\]

\vspace{0.5em}
\begin{block}{Final Solution}
\[
y = \frac{5}{4}e^{2x} - \frac{x^2}{2} - \frac{x}{2} - \frac{1}{4}
\]
\end{block}
\end{frame}

%--- Verification ---
\begin{frame}{Verification}
\textbf{Check ODE:} $\quad y = \dfrac{5}{4}e^{2x} - \dfrac{x^2}{2} - \dfrac{x}{2} - \dfrac{1}{4}$

\[
\frac{dy}{dx} = \frac{5}{2}e^{2x} - x - \frac{1}{2}
\]

\[
x^2 + 2y = x^2 + \frac{5}{2}e^{2x} - x^2 - x - \frac{1}{2} = \frac{5}{2}e^{2x} - x - \frac{1}{2} \checkmark
\]

\vspace{0.4em}
\textbf{Check IC:} $\quad y(0) = \dfrac{5}{4} - 0 - 0 - \dfrac{1}{4} = 1$ \checkmark

\vspace{0.5em}
\begin{block}{Conclusion}
\[
y = \frac{5}{4}e^{2x} - \frac{x^2}{2} - \frac{x}{2} - \frac{1}{4}
\]
\end{block}
\end{frame}

%--- RK4 Overview ---
\begin{frame}{RK4 Method: Overview}
The ODE is $f(x,y) = x^2 + 2y$.

\vspace{0.3em}
The \textbf{Runge-Kutta 4th order} update rule:
\[
k_1 = h\,f(x_n,\, y_n)
\]
\[
k_2 = h\,f\!\left(x_n + \tfrac{h}{2},\; y_n + \tfrac{k_1}{2}\right)
\]
\[
k_3 = h\,f\!\left(x_n + \tfrac{h}{2},\; y_n + \tfrac{k_2}{2}\right)
\]
\[
k_4 = h\,f\!\left(x_n + h,\; y_n + k_3\right)
\]
\[
y_{n+1} = y_n + \frac{1}{6}(k_1 + 2k_2 + 2k_3 + k_4)
\]
Starting from $x_0 = 0,\; y_0 = 1$, step size $h = 0.25$.
\end{frame}

import matplotlib.pyplot as plt

def f(x, y):
    return x**2 + 2*y

def rk4(f, x0, y0, xf, h):
    xs, ys = [x0], [y0]
    x, y = x0, y0
    while x < xf - 1e-10:
        k1 = f(x, y)
        k2 = f(x + h/2, y + h/2 * k1)
        k3 = f(x + h/2, y + h/2 * k2)
        k4 = f(x + h, y + h * k3)
